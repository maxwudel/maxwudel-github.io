teamzscores <- eastaverages %>% 
  mutate(
    FGZscore = as.numeric(scale(`FG%`, center = TRUE, scale = TRUE)),
    RebZscore = as.numeric(scale(RPG, center = TRUE, scale = TRUE)),
    AstZscore = as.numeric(scale(APG, center = TRUE, scale = TRUE)),
    TovZscore = as.numeric(scale(TOV, center = TRUE, scale = TRUE)),
    TotalZscore = FGZscore + RebZscore + AstZscore - TovZscore
  ) %>% 
  select(Team,FGZscore,RebZscore,AstZscore,TovZscore,TotalZscore)
finalscores <- teamzscores %>% arrange(desc(TotalZscore)) %>% select(Team,TotalZscore)
